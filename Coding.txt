install.packages("tseries")
install.packages("nortest")
install.packages("vars")
install.packages("forecast")
install.packages("lmtest")
require(tseries)
require(nortest)
require(vars)
require(forecast)
require(lmtest)
Rmf=scan("Rmf.txt")
Ytf=scan("Ytf.txt")
Rmf=ts(Rmf,start=2010,frequency=12)
Ytf=ts(Ytf,start=2010,frequency=12)
par(mfrow=c(2,1))
plot(Rmf)
plot(Ytf)
adf.test(Rmf)
adf.test(Ytf)
dRmf=diff(Rmf)
dYtf=diff(Ytf)
adf.test(dRmf)
adf.test(dYtf)
x=cbind(dRmf,dYtf)
VARselect(x)
y=VAR(x,p=1)
summary(y)
lmodel <- lm(Ytf ~ Rmf)
summary(lmodel)
grangertest(Ytf ~ Rmf, order = 1)
grangertest(Rmf ~ Ytf, order = 1)
